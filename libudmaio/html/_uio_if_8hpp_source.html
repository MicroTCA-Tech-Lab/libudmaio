119 template <typename access_width_t>
120 void _rd_mem(uint32_t offs, void* __restrict__ mem, size_t size) const {
121 access_width_t* __restrict__ ptr = reinterpret_cast<access_width_t*>(mem);
122 for (uint32_t i = 0; i < size; i += sizeof(access_width_t)) {
123 *ptr++ = _rd<access_width_t>(offs + i);

135 template <typename access_width_t>
136 void _wr_mem(uint32_t offs, const void* __restrict__ mem, size_t size) {
137 // volatile needed to prevent the compiler from optimizing away the memory read
138 const volatile access_width_t* ptr = reinterpret_cast<const volatile access_width_t*>(mem);
139 for (uint32_t i = 0; i < size; i += sizeof(access_width_t)) {
140 _wr<access_width_t>(offs + i, *ptr++);
141 }
142 }

151 template <typename C>
152 C _rd_reg(uint32_t offs) const {
153 static_assert(sizeof(C) != 0);
154 static_assert(sizeof(C) % sizeof(uint32_t) == 0);
155
156 C result;
157
158 if constexpr (sizeof(C) > sizeof(uint32_t) && sizeof(C) % sizeof(uint64_t) == 0) {
159 if (!_force_32bit) {
160 _rd_mem<uint64_t>(offs, reinterpret_cast<void*>(&result), sizeof(C));
161 } else {
162 _rd_mem<uint32_t>(offs, reinterpret_cast<void*>(&result), sizeof(C));
163 }
164 } else {
165 _rd_mem<uint32_t>(offs, reinterpret_cast<void*>(&result), sizeof(C));
166 }
167
168 return result;
169 }
170
178 template <typename C>
179 void _wr_reg(uint32_t offs, const C& value) {
180 static_assert(sizeof(C) != 0);
181 static_assert(sizeof(C) % sizeof(uint32_t) == 0);
182
183 if constexpr (sizeof(C) > sizeof(uint32_t) && sizeof(C) % sizeof(uint64_t) == 0) {
184 if (!_force_32bit) {
185 _wr_mem<uint64_t>(offs, reinterpret_cast<const void*>(&value), sizeof(C));
186 } else {
187 _wr_mem<uint32_t>(offs, reinterpret_cast<const void*>(&value), sizeof(C));
188 }
189 } else {
190 _wr_mem<uint32_t>(offs, reinterpret_cast<const void*>(&value), sizeof(C));
191 }
192
193 // Prevent compiler & CPU from re-ordering. Make sure the register is actually written here
194 __sync_synchronize();
195 }
