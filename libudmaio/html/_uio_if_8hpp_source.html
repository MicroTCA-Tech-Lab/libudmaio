22template <typename bus_type_t>
23class UioIf : public UioIfBase {
24 static_assert(sizeof(bus_type_t) == sizeof(uint32_t) || sizeof(bus_type_t) == sizeof(uint64_t),
25 "Bus access width must be 32 or 64 bits");
26
27 public:
28 using UioIfBusType = bus_type_t;
29 using UioIfBase::UioIfBase;
30 virtual ~UioIf(){};
31
32 protected:
33 inline volatile bus_type_t* _reg_ptr(uint32_t offs) const {
34 return reinterpret_cast<volatile bus_type_t*>(static_cast<uint8_t*>(_mem) +
35 static_cast<ptrdiff_t>(offs));
36 }
37
38 bus_type_t _rd(uint32_t offs) const {
39 bus_type_t tmp = *_reg_ptr(offs);
40 BOOST_LOG_SEV(_slg, blt::severity_level::trace)
41 << _log_name() << ": read at 0x" << std::hex << offs << " = 0x" << tmp << std::dec;
42 return tmp;
43 }
44
45 void _wr(uint32_t offs, bus_type_t data) {
46 BOOST_LOG_SEV(_slg, blt::severity_level::trace)
47 << _log_name() << ": write at 0x" << std::hex << offs << " = 0x" << data << std::dec;
48 *_reg_ptr(offs) = data;
49 }
50
51 void _rd_mem(uint32_t offs, void* __restrict__ mem, size_t size) const override {
52 assert((offs % sizeof(bus_type_t)) == 0);
53 assert((size % sizeof(bus_type_t)) == 0);
54 bus_type_t* __restrict__ ptr = reinterpret_cast<bus_type_t*>(mem);
55 for (uint32_t i = 0; i < size; i += sizeof(bus_type_t)) {
56 *ptr++ = _rd(offs + i);
57 }
58 }
59
60 void _wr_mem(uint32_t offs, const void* __restrict__ mem, size_t size) override {
61 assert((offs % sizeof(bus_type_t)) == 0);
62 assert((size % sizeof(bus_type_t)) == 0);
63 const bus_type_t* __restrict__ ptr = reinterpret_cast<const bus_type_t*>(mem);
64 for (uint32_t i = 0; i < size; i += sizeof(bus_type_t)) {
65 _wr(offs + i, *ptr++);
66 }
67 }
68};
