134 template <typename access_width_t>

136 // volatile needed to prevent the compiler from optimizing away the memory read
137 const volatile access_width_t* ptr = reinterpret_cast<const volatile access_width_t*>(mem);
138 for (uint32_t i = 0; i < size; i += sizeof(access_width_t)) {
139 _wr<access_width_t>(offs + i, *ptr++);

150 template <typename C>
151 C _rd_reg(uint32_t offs) const {
152 static_assert(sizeof(C) != 0);
153 static_assert(sizeof(C) % sizeof(uint32_t) == 0);
154
155 C result;
156 if constexpr (sizeof(C) > sizeof(uint32_t) && sizeof(C) % sizeof(uint64_t) == 0) {
157 if (!_force_32bit) {
158 _rd_mem<uint64_t>(offs, reinterpret_cast<void*>(&result), sizeof(C));
159 } else {
160 _rd_mem<uint32_t>(offs, reinterpret_cast<void*>(&result), sizeof(C));
161 }
162 } else {
163 _rd_mem<uint32_t>(offs, reinterpret_cast<void*>(&result), sizeof(C));
164 }
165
166 return result;
167 }

176 template <typename C>
177 void _wr_reg(uint32_t offs, const C& value) {
178 static_assert(sizeof(C) != 0);
179 static_assert(sizeof(C) % sizeof(uint32_t) == 0);
180
181 if constexpr (sizeof(C) > sizeof(uint32_t) && sizeof(C) % sizeof(uint64_t) == 0) {
182 if (!_force_32bit) {
183 _wr_mem<uint64_t>(offs, reinterpret_cast<const void*>(&value), sizeof(C));
184 } else {
185 _wr_mem<uint32_t>(offs, reinterpret_cast<const void*>(&value), sizeof(C));
186 }
187 } else {
188 _wr_mem<uint32_t>(offs, reinterpret_cast<const void*>(&value), sizeof(C));
189 }
190
191 // Prevent compiler & CPU from re-ordering. Make sure the register is actually written here
192 __sync_synchronize();
193 }
