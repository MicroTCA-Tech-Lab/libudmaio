72 template <typename C>
73 C _rd_reg(uint32_t offs) const {
74 static_assert(sizeof(C) != 0);
75 static_assert(sizeof(C) % sizeof(uint32_t) == 0);
76
77 C result;
78 if constexpr (sizeof(C) > sizeof(uint32_t) && sizeof(C) % sizeof(uint64_t) == 0) {
79 _rd_mem64(offs, reinterpret_cast<void*>(&result), sizeof(C));
80 } else {
81 _rd_mem32(offs, reinterpret_cast<void*>(&result), sizeof(C));
82 }
83
84 return result;
85 }

94 template <typename C>
95 void _wr_reg(uint32_t offs, const C& value) {
96 static_assert(sizeof(C) != 0);
97 static_assert(sizeof(C) % sizeof(uint32_t) == 0);
98
99 if constexpr (sizeof(C) > sizeof(uint32_t) && sizeof(C) % sizeof(uint64_t) == 0) {
100 _wr_mem64(offs, reinterpret_cast<const void*>(&value), sizeof(C));
101 } else {
102 _wr_mem32(offs, reinterpret_cast<const void*>(&value), sizeof(C));
103 }
104
105 // Prevent compiler & CPU from re-ordering. Make sure the register is actually written here
106 __sync_synchronize();
107 }

111 virtual int get_fd_int() const { return -1; }

116// Base class for mmap'ed hardware access
117template <typename max_access_width_t>
118class HwAccessorMmap : public HwAccessor {
119 public:
120 HwAccessorMmap(std::string dev_path, UioRegion region, uintptr_t mmap_offs)
121 : HwAccessor(), _region{region} {
122 BOOST_LOG_SEV(HwAccessor::_lg, bls::debug) << "dev name = " << dev_path;
123
124 _fd = ::open(dev_path.c_str(), O_RDWR);
125 if (_fd < 0) {
126 throw std::runtime_error("could not open " + dev_path);
127 }
128 BOOST_LOG_SEV(HwAccessor::_lg, bls::trace) << "fd = " << _fd << ", size = " << _region.size;
129
130 // create memory mapping
131 _mem = mmap(NULL, _region.size, PROT_READ | PROT_WRITE, MAP_SHARED, _fd, mmap_offs);
132 BOOST_LOG_SEV(HwAccessor::_lg, bls::trace)
133 << "mmap = 0x" << std::hex << mmap_offs << " -> 0x" << _mem << std::dec;
134
135 if (_mem == MAP_FAILED) {
136 throw std::runtime_error("mmap failed for uio " + dev_path);
137 }
138 }
139
140 virtual ~HwAccessorMmap() {
141 munmap(_mem, _region.size);
142 ::close(_fd);
143 }
144
145 protected:
146 int _fd;
147 void* _mem;
148 const UioRegion _region;
149
150 template <typename access_width_t>
151 inline volatile access_width_t* _mem_ptr(uint32_t offs) const {
152 return reinterpret_cast<access_width_t*>(static_cast<uint8_t*>(_mem) +
153 static_cast<ptrdiff_t>(offs));
154 }
155
156 uint32_t _rd32(uint32_t offs) const final override {
157 const uint32_t tmp = *_mem_ptr<uint32_t>(offs);
158 if (_debug_enable) {
159 BOOST_LOG_SEV(_lg, bls::trace)
160 << "0x" << std::hex << offs << " --> 0x" << std::setw(sizeof(uint32_t) * 2)
161 << std::setfill('0') << tmp << std::dec;
162 }
163 return tmp;
164 }
165
166 void _wr32(uint32_t offs, const uint32_t data) final override {
167 if (_debug_enable) {
168 BOOST_LOG_SEV(_lg, bls::trace)
169 << "0x" << std::hex << offs << " <-- 0x" << std::setw(sizeof(uint32_t) * 2)
170 << std::setfill('0') << data << std::dec;
171 }
172 *_mem_ptr<uint32_t>(offs) = data;
173 }
174
175 uint64_t _rd64(uint32_t offs) const final override {
176 if constexpr (sizeof(max_access_width_t) == sizeof(uint64_t)) {
177 // We support native 64-bit access
178 const uint64_t tmp = *_mem_ptr<uint64_t>(offs);
179 if (_debug_enable) {
180 BOOST_LOG_SEV(_lg, bls::trace)
181 << "0x" << std::hex << offs << " --> 0x" << std::setw(sizeof(uint64_t) * 2)
182 << std::setfill('0') << tmp << std::dec;
183 }
184 return tmp;
185 } else {
186 // Use default (32-bit) implementation
187 return HwAccessor::_rd64(offs);
188 }
189 }
190
191 void _wr64(uint32_t offs, const uint64_t data) final override {
192 if constexpr (sizeof(max_access_width_t) == sizeof(uint64_t)) {
193 // We support native 64-bit access
194 if (_debug_enable) {
195 BOOST_LOG_SEV(_lg, bls::trace)
196 << "0x" << std::hex << offs << " <-- 0x" << std::setw(sizeof(uint64_t) * 2)
197 << std::setfill('0') << data << std::dec;
198 }
199 *_mem_ptr<uint64_t>(offs) = data;
200 } else {
201 // Use default (32-bit) implementation
202 HwAccessor::_wr64(offs, data);
203 }
204 }
205
206 uint32_t wait_for_interrupt() final override {
207 const int fd_int = get_fd_int();
208 uint32_t irq_count;
209 BOOST_LOG_SEV(_lg, bls::trace) << "wait for interrupt ...";
210 if (fd_int < 0) {
211 BOOST_LOG_SEV(_lg, bls::fatal) << "fd_int not set";
212 }
213 int rc = read(fd_int, &irq_count, sizeof(irq_count));
214 BOOST_LOG_SEV(_lg, bls::trace)
215 << "interrupt received, rc = " << rc << ", irq count = " << irq_count;
216 return irq_count;
217 }
218
219 public:
220 UioRegion get_phys_region() const final override { return _region; }
221 void* get_virt_mem() const final override { return _mem; }
222};

224// Hardware accessor for XDMA. Can support either 32 or 64 bit access
225template <typename max_access_width_t>
226class HwAccessorXdma : public HwAccessorMmap<max_access_width_t> {
227 int _fd_int;
228
229 public:
230 HwAccessorXdma(std::string xdma_path,
231 std::string evt_dev,
232 UioRegion region,
233 uintptr_t pcie_offs)
234 : HwAccessorMmap<max_access_width_t>(xdma_path + "/user",
235 {region.addr | pcie_offs, region.size},
236 region.addr) {
237 if (!evt_dev.empty()) {
238 const auto evt_path = xdma_path + "/" + evt_dev;
239 _fd_int = ::open(evt_path.c_str(), O_RDWR);
240
241 if (_fd_int < 0) {
242 throw std::runtime_error("could not open " + evt_path);
243 }
244 BOOST_LOG_SEV(HwAccessor::_lg, bls::trace) << "fd_int = " << _fd_int;
245 } else {
246 _fd_int = -1;
247 }
248 }
249
250 virtual ~HwAccessorXdma() {
251 if (_fd_int > 0) {
252 ::close(_fd_int);
253 }
254 }
255
256 int get_fd_int() const final override { return _fd_int; }
257}; // namespace udmaio
